  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"openai_gpt-oss-20b-MXFP4.gguf\",

    \"temperature\": 0.1,
    \"max_tokens\": 200,
    \"stop\": [\"Note:\", \"解説\", \"Sure\", \"So \", \"申し訳ありません\", \"以下の\"]
  }" | jq -r '.choices[0].text'
